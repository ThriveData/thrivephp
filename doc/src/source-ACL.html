    4: /**
    5: * Query access control for users and roles.
    6: *
    7: * There are four database tables involved in access control:
    8: * - roles table: The roles table is a list of roles that can be assigned to users.
    9: * - users_roles: The users_roles table stores all the roles assigned to users.
    10: * - acl_keys: The acl_keys table is a list of items that represent things for which
    11: * access should be controlled. Most keys represent a database tables or data objects.
    12: * - acl_permissions: The acl_permissions table stores the permissions granted permissions
    13: * granted for a key to a role.
    14: *
    15: * Example: Johnny Appleseed is a sales manager and needs permissions to create, edit, and
    16: * delete estimates for himself and his team. Johnny will need to be assigned the Sales Manager role.
    17: * And the Sales Manager role should be given access to the key `data.estimates` and permission grants
    18: * of `select`, `update`, `create`, `delete`, and `manager`.
    19: *
    20: */
    21: class ACL
    22: {
    23: const none = 0;
    24: const select = 1; // User has read/view/select permission for items belonging to himself
    25: const update = 2; // User has write/save/update permission for items belonging to himeself
    26: const create = 4; // User has write/create/insert permission for items belonging to himself
    27: const delete = 8; // User has delete permission for items belonging to himself
    28: const manager = 16; // User has the above permissions for items belonging to users for who he is a manager
    29: const admin = 32; // User has the above permissions for all items
    30: const grant = 64; // User has permission to grant to other users
    31: const revoke = 128; // User has permission to revoke from other users

    33: /**
    34: * Check if `$userId` is granted `$permissions` for `$key`.
    35: *
    36: * The call `check('data.estimates', ACL::select)` will check if the current user
    37: * can view estimates for himself only. The call `check('data.estimates', ACL::select|ACL::manager)`
    38: * will check if the current user can view estimates for himself and anyone for who he is a manager.
    39: *
    40: * The expression `ACL::select|ACL::manager`, where `|` is the binary OR operator, means the
    41: * user needs to have both select and manager grants. It can also be expressed like this:
    42: *
    43: * ```
    44: * Binary: Decimal:
    45: * 10000 1
    46: * | 00001 + 16
    47: * ----- --
    48: * 10001 17
    49: * ```
    50: *
    51: * @param string $key The key from the `acl_keys` database table.
    52: *
    53: * @param int $permissions The granted permissions for `$key`. Possible permissions are:
    54: * - ACL::none
    55: * - ACL::select
    56: * - ACL::update
    57: * - ACL::create
    58: * - ACL::delete
    59: * - ACL::manager
    60: * - ACL::admin
    61: *
    62: * @param uuid $userId The user's ID from the `users` database table.
    63: */
    64: static function check($key, $permissions, $userId=null)
    65: {
    66: $userId = $userId ?? $_SESSION['user']['id'] ?? null;
    67: if (is_null($userId)) throw new PermissionsException('could not get user ID to check permissions');
    68:
    69: // Check permissions with session (avoids potentially hundreds of database lookups)
    70: if ($session = ($_SESSION['user']['permissions'] ?? null)):

    72: $grants = $session->acl->$key ?? null;
    73:
    74: if ($superuser):
    75: return true;
    76: endif;
    77:
    78: if (($grants & $permissions) == $permissions):
    79: return true;
    80: endif;
    81:
    82: // Check permissions with database lookup
    83: else:

    85: $count = DB::query("SELECT count(*) AS count FROM users WHERE id=$1 AND superuser IS TRUE", $userId)

    100: if ($count > 0)
    101: return true;
    102: endif;
    103:
    104: return false;
    105: }
